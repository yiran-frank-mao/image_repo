\newcommand{\braido}{
\begin{tikzpicture}[baseline=(current bounding box.center)]
  % Named braid, rotated so it runs left-to-right
  \pic[
    rotate=90,
    name=b,
    braid/.cd,
    number of strands=2,
    width=4.5mm, % smaller than the default spacing
    crossing height=4.5mm, % shorter braid body
    border height=0.75mm,   % shorter strand ends
    % every strand/.style={thick},
  ] {braid={s_1}};

  % dots on the left side (starts)
  \foreach \i in {1,2}
    \fill (b-\i-s) circle (1.8pt);

  % dots on the right side (ends)
  \foreach \i in {1,2}
    \fill (b-\i-e) circle (1.8pt);
\end{tikzpicture}
}
